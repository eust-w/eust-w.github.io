Definition for singly-linked list.

class ListNode: def init(self, x): self.val = x self.next = None

class Solution: def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode: prenode = ListNode(0) lastnode = prenode val = 0 while val or l1 or l2: val, cur = divmod(val + (l1.val if l1 else 0) + (l2.val if l2 else 0), 10) lastnode.next = ListNode(cur) lastnode = lastnode.next l1 = l1.next if l1 else None l2 = l2.next if l2 else None return prenode.next
